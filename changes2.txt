return Dialog(
  insetPadding: const EdgeInsets.symmetric(horizontal: 12, vertical: 16),
  shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(16)),
  child: ConstrainedBox(
    constraints: BoxConstraints(
      // ancho máximo igual que tenías (responsive)
      maxWidth: maxDialogWidth > 420 ? 420 : maxDialogWidth,
    ),
    child: Material(
      color: Colors.white,
      borderRadius: BorderRadius.circular(16),
      child: Padding(
        padding: const EdgeInsets.fromLTRB(16, 14, 16, 12),
        child: Column(
          mainAxisSize: MainAxisSize.min,
          children: [
            // TITLE
            const Align(
              alignment: Alignment.centerLeft,
              child: Text(
                "Create appointment",
                style: TextStyle(fontWeight: FontWeight.w800, fontSize: 18),
              ),
            ),
            const SizedBox(height: 10),

            // CONTENT (SCROLL)
            Flexible(
              child: SingleChildScrollView(
                child: Form(
                  key: formKey,
                  child: Column(
                    mainAxisSize: MainAxisSize.min,
                    children: [
                        // MODE TOGGLE
                        Container(
                          padding: const EdgeInsets.all(6),
                          decoration: BoxDecoration(
                            color: Colors.grey.withOpacity(0.10),
                            borderRadius: BorderRadius.circular(14),
                          ),
                          child: Row(
                            children: [
                              Expanded(
                                child: InkWell(
                                  borderRadius: BorderRadius.circular(12),
                                  onTap: () {
                                    setLocalState(() {
                                      existingClientMode = true;
                                      // limpia new client
                                      firstNameCtrl.clear();
                                      lastNameCtrl.clear();
                                      countryCtrl.clear();
                                      phoneCtrl.clear();
                                      instagramCtrl.clear();
                                    });
                                  },
                                  child: Container(
                                    padding: const EdgeInsets.symmetric(
                                        vertical: 10),
                                    decoration: BoxDecoration(
                                      borderRadius: BorderRadius.circular(12),
                                      color: existingClientMode
                                          ? const Color(0xff721c80)
                                          : Colors.transparent,
                                    ),
                                    child: Center(
                                      child: Text(
                                        "Existing",
                                        overflow: TextOverflow.ellipsis,
                                        style: TextStyle(
                                          color: existingClientMode
                                              ? Colors.white
                                              : Colors.black87,
                                          fontWeight: FontWeight.w700,
                                        ),
                                      ),
                                    ),
                                  ),
                                ),
                              ),
                              const SizedBox(width: 6),
                              Expanded(
                                child: InkWell(
                                  borderRadius: BorderRadius.circular(12),
                                  onTap: () {
                                    setLocalState(() {
                                      existingClientMode = false;
                                      selectedClientId = '';
                                      selectedClientData = null;
                                      clientSearchCtrl.clear();
                                    });
                                  },
                                  child: Container(
                                    padding: const EdgeInsets.symmetric(
                                        vertical: 10),
                                    decoration: BoxDecoration(
                                      borderRadius: BorderRadius.circular(12),
                                      color: !existingClientMode
                                          ? const Color(0xff721c80)
                                          : Colors.transparent,
                                    ),
                                    child: Center(
                                      child: Text(
                                        "New",
                                        overflow: TextOverflow.ellipsis,
                                        style: TextStyle(
                                          color: !existingClientMode
                                              ? Colors.white
                                              : Colors.black87,
                                          fontWeight: FontWeight.w700,
                                        ),
                                      ),
                                    ),
                                  ),
                                ),
                              ),
                            ],
                          ),
                        ),

                        const SizedBox(height: 12),

                        if (existingClientMode) ...[
                          TextFormField(
                            controller: clientSearchCtrl,
                            decoration: const InputDecoration(
                              labelText: "Search (name / phone / instagram)",
                              border: OutlineInputBorder(),
                              prefixIcon: Icon(Icons.search),
                            ),
                            onChanged: (_) => setLocalState(() {}),
                          ),
                          const SizedBox(height: 10),

                          // ─────────────────────────────
                          // PUNTO 1: CAJA FIJA 3 FILAS + SCROLL
                          // ─────────────────────────────
                          StreamBuilder<QuerySnapshot>(
                            stream: FirebaseFirestore.instance
                                .collection('clients')
                                .orderBy('updatedAt', descending: true)
                                .limit(120)
                                .snapshots(),
                            builder: (context, snap) {
                              if (snap.connectionState ==
                                  ConnectionState.waiting) {
                                return const SizedBox(
                                  height: 28,
                                  child: Center(
                                    child: LinearProgressIndicator(),
                                  ),
                                );
                              }
                              if (!snap.hasData || snap.data!.docs.isEmpty) {
                                return const SizedBox.shrink();
                              }

                              final q = clientSearchCtrl.text
                                  .trim()
                                  .toLowerCase();
                              final docs = snap.data!.docs;

                              List<QueryDocumentSnapshot> filtered;

                              if (q.isEmpty) {
                                // recientes si no escribe nada
                                filtered = docs.take(10).toList();
                              } else {
                                filtered = docs.where((d) {
                                  final data =
                                      d.data() as Map<String, dynamic>? ?? {};
                                  final search =
                                      (data['search'] as List<dynamic>?)
                                              ?.map((e) => e
                                                  .toString()
                                                  .toLowerCase())
                                              .toList() ??
                                          [];

                                  final first = (data['firstName'] ?? '')
                                      .toString()
                                      .toLowerCase();
                                  final last = (data['lastName'] ?? '')
                                      .toString()
                                      .toLowerCase();
                                  final ig = (data['instagram'] ?? '')
                                      .toString()
                                      .toLowerCase();

                                  final nameMatch = ('$first $last')
                                          .contains(q) ||
                                      first.contains(q) ||
                                      last.contains(q);

                                  final igMatch = ig.contains(
                                      q.replaceAll('@', ''));

                                  final searchMatch = search
                                      .any((t) => t.contains(q));

                                  return nameMatch || igMatch || searchMatch;
                                }).toList();
                              }

                              if (filtered.isEmpty) {
                                return Text(
                                  "No matches",
                                  style: TextStyle(color: Colors.grey[700]),
                                );
                              }

                              const tileHeight = 56.0;
                              const maxVisible = 3;
                              const boxHeight = tileHeight * maxVisible;

                              return Container(
                                padding: const EdgeInsets.all(8),
                                decoration: BoxDecoration(
                                  color: Colors.grey.withOpacity(0.06),
                                  borderRadius: BorderRadius.circular(12),
                                  border: Border.all(color: Colors.black12),
                                ),
                                child: SizedBox(
                                  height: boxHeight,
                                  child: Scrollbar(
                                    thumbVisibility: true,
                                    child: ListView.separated(
                                      itemCount: filtered.length,
                                      separatorBuilder: (_, __) =>
                                          const Divider(height: 1),
                                      itemBuilder: (_, i) {
                                        final d = filtered[i];
                                        final first =
                                            (d['firstName'] ?? '').toString();
                                        final last =
                                            (d['lastName'] ?? '').toString();

                                        final ctry = d['country'];
                                        final ph = d['phone'];

                                        final phoneText = (ctry is num &&
                                                ph is num &&
                                                ctry > 0 &&
                                                ph > 0)
                                            ? "+${ctry.toInt()} ${ph.toInt()}"
                                            : "";

                                        final ig =
                                            (d['instagram'] ?? '').toString();

                                        final subtitle = [
                                          phoneText.isNotEmpty ? phoneText : "",
                                          ig.isNotEmpty ? "@$ig" : "",
                                        ]
                                            .where((x) =>
                                                x.trim().isNotEmpty)
                                            .join(" • ");

                                        return ListTile(
                                          dense: true,
                                          contentPadding: EdgeInsets.zero,
                                          title: Text(
                                            "$first $last",
                                            maxLines: 1,
                                            overflow: TextOverflow.ellipsis,
                                          ),
                                          subtitle: Text(
                                            subtitle,
                                            maxLines: 1,
                                            overflow: TextOverflow.ellipsis,
                                          ),
                                          trailing: const Icon(
                                            Icons.check_circle_outline,
                                            size: 18,
                                          ),
                                          onTap: () =>
                                              loadClient(d.id, setLocalState),
                                        );
                                      },
                                    ),
                                  ),
                                ),
                              );
                            },
                          ),

                          if (selectedClientData != null) ...[
                            const SizedBox(height: 10),
                            Container(
                              width: double.infinity,
                              padding: const EdgeInsets.all(10),
                              decoration: BoxDecoration(
                                color: const Color(0xff721c80)
                                    .withOpacity(0.08),
                                borderRadius: BorderRadius.circular(12),
                                border: Border.all(
                                  color: const Color(0xff721c80)
                                      .withOpacity(0.2),
                                ),
                              ),
                              child: Column(
                                crossAxisAlignment: CrossAxisAlignment.start,
                                children: [
                                  Text(
                                    clientNamePreview,
                                    maxLines: 1,
                                    overflow: TextOverflow.ellipsis,
                                    style: const TextStyle(
                                        fontWeight: FontWeight.w800),
                                  ),
                                  const SizedBox(height: 4),
                                  Text(
                                    contactPreview,
                                    maxLines: 1,
                                    overflow: TextOverflow.ellipsis,
                                    style: TextStyle(color: Colors.grey[700]),
                                  ),
                                ],
                              ),
                            ),
                          ],
                        ],

                        if (!existingClientMode) ...[
                          TextFormField(
                            controller: firstNameCtrl,
                            decoration: const InputDecoration(
                              labelText: "First name",
                              border: OutlineInputBorder(),
                            ),
                            validator: (v) => (v == null || v.trim().isEmpty)
                                ? "First name is required"
                                : null,
                          ),
                          const SizedBox(height: 10),
                          TextFormField(
                            controller: lastNameCtrl,
                            decoration: const InputDecoration(
                              labelText: "Last name",
                              border: OutlineInputBorder(),
                            ),
                            validator: (v) => (v == null || v.trim().isEmpty)
                                ? "Last name is required"
                                : null,
                          ),
                          const SizedBox(height: 10),
                          TextFormField(
                            controller: countryCtrl,
                            keyboardType: TextInputType.number,
                            decoration: const InputDecoration(
                              labelText: "Country code (e.g. 351, 34)",
                              border: OutlineInputBorder(),
                              prefixIcon: Icon(Icons.flag_outlined),
                            ),
                          ),
                          const SizedBox(height: 10),
                          TextFormField(
                            controller: phoneCtrl,
                            keyboardType: TextInputType.number,
                            decoration: const InputDecoration(
                              labelText: "Phone (optional)",
                              border: OutlineInputBorder(),
                              prefixIcon: Icon(Icons.phone),
                            ),
                          ),
                          const SizedBox(height: 10),
                          TextFormField(
                            controller: instagramCtrl,
                            decoration: const InputDecoration(
                              labelText: "Instagram (optional)",
                              border: OutlineInputBorder(),
                              prefixIcon: Icon(Icons.alternate_email),
                            ),
                          ),
                          const SizedBox(height: 8),
                          Text(
                            "Phone or Instagram is required",
                            style: TextStyle(
                                color: Colors.grey[700], fontSize: 12),
                          ),
                        ],

                        const SizedBox(height: 12),

                        // PROCEDURE
                        StreamBuilder<QuerySnapshot>(
                          stream: FirebaseFirestore.instance
                              .collection('services')
                              .snapshots(),
                          builder: (context, snap) {
                            if (snap.connectionState ==
                                ConnectionState.waiting) {
                              return const Padding(
                                padding: EdgeInsets.symmetric(vertical: 10),
                                child: LinearProgressIndicator(),
                              );
                            }
                            if (snap.hasError ||
                                !snap.hasData ||
                                snap.data!.docs.isEmpty) {
                              return const Text("No services found");
                            }
                            final docs = snap.data!.docs;

                            return DropdownButtonFormField<String>(
                              value: selectedServiceId,
                              isExpanded: true,
                              decoration: const InputDecoration(
                                labelText: "Procedure",
                                border: OutlineInputBorder(),
                                isDense: true,
                                contentPadding: EdgeInsets.symmetric(
                                    horizontal: 12, vertical: 12),
                              ),
                              items: docs.map((d) {
                                final data =
                                    d.data() as Map<String, dynamic>? ?? {};
                                final nameKey =
                                    (data['name'] ?? d.id).toString();
                                final label =
                                    trServiceOrAddon(context, nameKey);

                                return DropdownMenuItem<String>(
                                  value: d.id,
                                  child: Text(
                                    label,
                                    maxLines: 1,
                                    overflow: TextOverflow.ellipsis,
                                  ),
                                );
                              }).toList(),
                              onChanged: (val) async {
                                selectedServiceId = val;
                                selectedTypeKey = null;
                                selectedTypeData = null;

                                if (val != null) {
                                  final doc = await FirebaseFirestore.instance
                                      .collection('services')
                                      .doc(val)
                                      .get();
                                  selectedServiceData = doc.data();
                                } else {
                                  selectedServiceData = null;
                                }
                                setLocalState(() {});
                              },
                              validator: (v) =>
                                  (v == null || v.isEmpty)
                                      ? "Procedure is required"
                                      : null,
                            );
                          },
                        ),

                        // PUNTO 3: TYPES dropdown si el servicio lo requiere
                        if (types.isNotEmpty) ...[
                          const SizedBox(height: 10),
                          DropdownButtonFormField<String>(
                            value: selectedTypeKey,
                            isExpanded: true,
                            decoration: const InputDecoration(
                              labelText: "Type / Size",
                              border: OutlineInputBorder(),
                              isDense: true,
                              contentPadding: EdgeInsets.symmetric(
                                  horizontal: 12, vertical: 12),
                            ),
                            items: types.map((t) {
                              final key = (t['key'] ?? '').toString();
                              final nameKey =
                                  (t['nameKey'] ?? key).toString();
                              final label =
                                  trServiceOrAddon(context, nameKey);

                              return DropdownMenuItem<String>(
                                value: key,
                                child: Text(
                                  label,
                                  maxLines: 1,
                                  overflow: TextOverflow.ellipsis,
                                ),
                              );
                            }).toList(),
                            onChanged: (v) {
                              selectedTypeKey = v;
                              selectedTypeData = types.firstWhere(
                                (t) =>
                                    (t['key'] ?? '').toString() == v,
                                orElse: () => types.first,
                              );
                              setLocalState(() {});
                            },
                            validator: (v) =>
                                (v == null || v.isEmpty)
                                    ? "Type is required"
                                    : null,
                          ),
                        ],

                        const SizedBox(height: 12),

                        // TIME (PUNTO 4: 5 en 5)
                        InkWell(
                          onTap: () async {
                            final initial = selectedTime ?? TimeOfDay.now();
                            final t = await _pickTime5(ctx, initial);
                            if (t != null) {
                              setLocalState(() => selectedTime = t);
                            }
                          },
                          borderRadius: BorderRadius.circular(12),
                          child: Container(
                            width: double.infinity,
                            padding: const EdgeInsets.symmetric(
                                horizontal: 12, vertical: 12),
                            decoration: BoxDecoration(
                              borderRadius: BorderRadius.circular(12),
                              border: Border.all(color: Colors.black26),
                            ),
                            child: Row(
                              children: [
                                const Icon(Icons.schedule, size: 18),
                                const SizedBox(width: 10),
                                Flexible(
                                  child: Text(
                                    selectedTime == null
                                        ? "Select time"
                                        : "${selectedTime!.hour.toString().padLeft(2, '0')}:${selectedTime!.minute.toString().padLeft(2, '0')}",
                                    maxLines: 1,
                                    overflow: TextOverflow.ellipsis,
                                    style: const TextStyle(
                                        fontWeight: FontWeight.w600),
                                  ),
                                ),
                                const Spacer(),
                                const Icon(Icons.edit,
                                    size: 16, color: Colors.black54),
                              ],
                            ),
                          ),
                        ),

                        const SizedBox(height: 10),

                        if (selectedServiceData != null)
                          Row(
                            children: [
                              Flexible(
                                child: Text(
                                  "Price: €${price.toStringAsFixed(0)}",
                                  maxLines: 1,
                                  overflow: TextOverflow.ellipsis,
                                  style: const TextStyle(
                                      fontWeight: FontWeight.w800),
                                ),
                              ),
                              const Spacer(),
                              Text(
                                durationMin > 0 ? "Time: ${durationMin}m" : "Time: —",
                                style: const TextStyle(
                                  color: Colors.black54,
                                  fontWeight: FontWeight.w600,
                                ),
                              ),
                            ],
                          ),
                      ],
                  ),
                ),
              ),
            ),

            const SizedBox(height: 10),

            // ACTIONS (buttons)
            Row(
              children: [
                Expanded(
                  child: TextButton(
                    onPressed: saving ? null : () => Navigator.pop(ctx),
                    child: const Text("Cancel"),
                  ),
                ),
                const SizedBox(width: 10),
                Expanded(
                  child: ElevatedButton(
                    style: ElevatedButton.styleFrom(
                      backgroundColor: const Color(0xff721c80),
                      shape: RoundedRectangleBorder(
                        borderRadius: BorderRadius.circular(12),
                      ),
                    ),
                    onPressed: saving ? null : () async {
                      // ⬇️ pega aquí TODO tu onPressed actual del botón Create
                    },
                    child: saving
                        ? const SizedBox(
                            width: 18,
                            height: 18,
                            child: CircularProgressIndicator(
                              strokeWidth: 2,
                              color: Colors.white,
                            ),
                          )
                        : const Text(
                            "Create",
                            style: TextStyle(
                              color: Colors.white,
                              fontWeight: FontWeight.w800,
                            ),
                          ),
                  ),
                ),
              ],
            ),
          ],
        ),
      ),
    ),
  ),
);
